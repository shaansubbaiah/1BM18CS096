.MODEL SMALL

DISPLAY MACRO MSG
LEA DX, MSG
MOV AH, 09H
INT 21H
ENDM

.DATA
    MSG1 DB 0DH, 0AH, "ENTER A STRING:$"
    MSG2 DB 0DH, 0AH, "REVERSE STRING:$"
    MSG3 DB 0DH, 0AH, "PALINDROME!$"
    MSG4 DB 0DH, 0AH, "NOT PALINDROME!$"
    STRING DB 80H DUP(?)
    RSTRING DB 80H DUP(?)

.CODE
    MOV AX, @DATA
    MOV DS, AX

    DISPLAY MSG1

    MOV SI, OFFSET STRING
    XOR CL, CL

AGAIN:
    MOV AH, 01H
    INT 21H

    CMP AL, 0DH
    JE NEXT
    MOV [SI], AL
    INC SI
    INC CL
    JMP AGAIN

NEXT:
    MOV [SI], BYTE PTR '$'
    DEC SI
    
    MOV DI, OFFSET RSTRING
    MOV CH, CL

COPYY:
    MOV AL, [SI]
    MOV [DI], AL
    INC DI
    DEC SI
    DEC CH
    CMP CH, 00
    JNZ COPYY

    MOV [DI], BYTE PTR '$'
    DEC DI
    
    DISPLAY MSG2
    DISPLAY RSTRING

    ;CHECK IF PALIN
    MOV SI, OFFSET STRING
    MOV DI, OFFSET RSTRING
CHECKS:
    MOV AL, [SI]
    CMP AL, [DI]
    JNE FAIL
    INC SI
    INC DI
    DEC CL
    CMP CL, 00
    JNZ CHECKS
    JMP SUCCESS

FAIL:
    DISPLAY MSG4
    JMP FINAL

SUCCESS:
    DISPLAY MSG3

FINAL:
    MOV AH, 4CH
    INT 21H
END
