.MODEL SMALL
.DATA
    ST1 DB 'TEMPLE$'
    LEN EQU $-ST1
    KEY DB 'E'

.CODE
    MOV AX, @DATA  
    MOV DS, AX
    MOV ES, AX

    LEA DI, ST1
    MOV AL, KEY
    MOV CX, LEN-1

    CLD

    REPNE SCASB
    JZ AGAIN
AGAIN:    
    MOV ES:[DI-1], BYTE PTR 'A' ;DI GETS INCREMENTED AFTER SCASB TO NEXT CHARACTER

    MOV AH, 4CH
    INT 21H
END
